    JAVASCRIPT CODE

    <script>
        function appendValue(val) {
            document.getElementById("result").value += val;
        }

        function clearDisplay() {
            document.getElementById("result").value = '';
        }

        function deleteLast() {
            let val = document.getElementById("result").value;
            document.getElementById("result").value = val.slice(0, -1);
        }

        function calculate() {
            try {
                document.getElementById("result").value = eval(document.getElementById("result").value);
            } catch {
                document.getElementById("result").value = "Error";
            }
        }

        function checkPrime() {
            let n = parseInt(document.getElementById("result").value);
            if (isNaN(n)) {
                document.getElementById("result").value = "Invalid input";
                return;
            }
            let isPrime = true;
            if (n <= 1) isPrime = false;
            for (let i = 2; i <= n / 2; i++) {
                if (n % i === 0) {
                    isPrime = false;
                    break;
                }
            }
            document.getElementById("result").value = isPrime ? "Prime Number" : "Not a Prime";
        }

        function calculateFactorial() {
            let n = parseInt(document.getElementById("result").value);
            if (isNaN(n) || n < 0) {
                document.getElementById("result").value = "Invalid input";
                return;
            }
            let fact = 1;
            for (let i = 2; i <= n; i++) {
                fact *= i;
            }
            document.getElementById("result").value = fact;
        }
    </script>
